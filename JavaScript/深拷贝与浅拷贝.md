浅拷贝：b 复制 a  ，改变 a，b 也变了
深拷贝：b 复制 a  ，改变 a，b 没变
```javascript
let a = 1;
let b = a;
a = 2;
console.log(b); // 1
修改 a 的值，b 不变


let obj = {
  a: 1,
};
let b = obj;
obj.a = 2;
console.log(b); //{a: 2}
修改了 a 的值，b 也变了
```
# 两层深拷贝
```javascript
let obj = {
  a: 1,
  b: {
    c: 3,
    d: {
      e: 5,
    },
  },
};

function deepClone(obj) {
  let cloneObj = {};
  for (let i in obj) {
    if (typeof obj[i] === "object" && obj[i] !== null) {
      cloneObj[i] = {};
      for (let j in obj[i]) {
        cloneObj[i][j] = obj[i][j];
      }
    } else {
      cloneObj[i] = obj[i];
    }
  }
  return cloneObj;
}

let obj1 = deepClone(obj);
obj.b.c = 10;
console.log(obj1);
```
# 三层深拷贝
```javascript
function deepClone(obj) {
  let cloneObj = {};
  
  for (let i in obj) {
    if (typeof obj[i] === "object" && obj[i] !== null) {
      cloneObj[i] = {};
      for (let j in obj[i]) {
        if (typeof obj[i][j] === "object" && obj[i][j] !== null) {
          cloneObj[i][j] = {};
          for (let k in obj[i][j]) {
            cloneObj[i][j][k] = obj[i][j][k];
          }
        } else {
          cloneObj[i][j] = obj[i][j];
        }
      }
    } else {
      cloneObj[i] = obj[i];
    }
  }
  
  return cloneObj;
}

let obj1 = deepClone(obj);
obj.b.d.e = 10;
console.log(obj1);
```
# 递归
先判断 obj 的 key 是不是对象，是对象的话在判断是不是数组
```javascript
let obj = {
  a: 1,
  b: {
    c: 3,
    d: {
      e: 5,
      f: [1, 2, 3, 4, 5, 6],
    },
  },
};

function deepClone(obj, cloneObj) {
  var cloneObj = cloneObj || {};

  for (var i in obj) {
    if (typeof obj[i] === "object" && obj[i] !== null) { 
      cloneObj[i] = Array.isArray(obj[i]) ? [] : {}; // 判断是不是数组
      deepClone(obj[i], cloneObj[i]);
    } else {
      cloneObj[i] = obj[i];
    }
  }

  return cloneObj;
}

var obj1 = deepClone(obj);
obj.b.d.e = 10;
obj.b.d.f.push(100);
console.log(obj1);
```
三种方式判断是数组还是对象，可以在上面代码中替换
```javascript
cloneObj[i] = Array.isArray(obj[i]) ? [] : {};

cloneObj[i] = obj[i] instanceof Array ? [] : {};

cloneObj[i] = Object.prototype.toString.call(obj[i]) === "[object Array]" ? [] : {};
```
